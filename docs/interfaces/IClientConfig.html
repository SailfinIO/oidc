Interface IClientConfig

Represents the configuration for an OAuth/OIDC client.

The IClientConfig interface defines the parameters required to initialize and manage the client's authentication behavior, including session management, logging, and additional options for advanced use cases.

interface IClientConfig {
    acrValues?: string | string[];
    additionalParams?: Record<string, any>;
    clientId: string;
    clientSecret?: string;
    clockSkew?: number;
    discoveryUrl: string;
    display?: Display;
    grantType?: GrantType;
    issuer?: string;
    logging?: { logger?: ILogger; logLevel?: LogLevel };
    maxAge?: number;
    nonce?: string;
    pkce?: boolean;
    pkceMethod?: PkceMethod;
    postLogoutRedirectUri?: string;
    prompt?: LoginPrompt;
    redirectUri: string;
    responseMode?: ResponseMode;
    responseType?: ResponseType;
    retryAttempts?: number;
    scopes: string[] | Scopes[];
    session?: {
        cookie?: { name?: string; options?: CookieOptions; secret?: string };
        mechanism?: StorageMechanism;
        options?: StoreOptions;
        store?: ISessionStore;
        ttl?: number;
        useSilentRenew?: boolean;
    };
    state?: string;
    timeout?: number;
    tokenRefreshThreshold?: number;
    uiLocales?: UILocales[];
}
